import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset
df = pd.read_csv(r'data/e_commerce_dataset.csv')

# Display the first few rows of the dataset to understand its structure
df.head()

# Basic statistics about the dataset
df.describe()

# Check for missing values
df.isnull().sum()

# Basic info about data types
df.info()


# Convert 'OrderDate' to datetime format if it's not already
df['OrderDate'] = pd.to_datetime(df['OrderDate'])

# Group by date and sum the total sales
sales_over_time = df.groupby('OrderDate')['TotalPrice'].sum()

# Plot sales over time
plt.figure(figsize=(10,6))
sales_over_time.plot()
plt.title('Total Sales Over Time')
plt.ylabel('Total Sales ($)')
plt.xlabel('Order Date')
plt.show()


# Group by ProductCategory and sum the total sales
category_sales = df.groupby('ProductCategory')['TotalPrice'].sum()

# Plot the total sales by product category
plt.figure(figsize=(8,6))
category_sales.plot(kind='bar', color='skyblue')
plt.title('Total Sales by Product Category')
plt.ylabel('Total Sales ($)')
plt.show()

# Group by CustomerID and sum their total purchases
top_customers = df.groupby('CustomerID')['TotalPrice'].sum().sort_values(ascending=False).head(10)

# Plot the top customers
plt.figure(figsize=(8,6))
top_customers.plot(kind='bar', color='orange')
plt.title('Top 10 Customers by Total Spend')
plt.ylabel('Total Spend ($)')
plt.show()

df.to_csv('cleaned_ecommerce_data.csv', index=False)
